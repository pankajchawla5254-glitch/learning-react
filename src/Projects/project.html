File 01 — 1-state.jsx
useState Basics
Learned how useState holds a value and re-renders the component when updated. Used setName to swap between names on button click.
useState event handler re-render 
import { useState } from "react";

function State() {
  const [name, setName] = useState("Pankaj");

  const handleClick = () => {
    setName("Rohit"); // triggers re-render
  };

  return (
    <div>
      <h2>{name}</h2>
      <button onClick={handleClick}>Change</button>
    </div>
  );
}

useEffect Hook

useEffectdependency arrayside effects 
// Runs after render when [counter] or [data] changes

  counterFunction();
}, [counter, data]);

// [val] = runs when val changes
// no array = runs on every render
